# patch_gpt_inference.py
import re

file_path = r"C:\HEAI\venv\lib\site-packages\TTS\tts\layers\xtts\gpt_inference.py"

with open(file_path, "r", encoding="utf-8") as f:
    content = f.read()

original = content

# 1. GenerationMixin'i import'a ekle
# "from transformers import ..." satırını bul ve GenerationMixin ekle
if "GenerationMixin" not in content:
    # transformers import satırını bul
    content = re.sub(
        r'(from transformers import\s+)([^\n]+)',
        lambda m: m.group(0) if "GenerationMixin" in m.group(0) 
                  else m.group(1) + m.group(2).rstrip() + ", GenerationMixin",
        content,
        count=1
    )
    
    # Eğer yukarıdaki çalışmazsa, ilk transformers import'undan sonra ekle
    if "GenerationMixin" not in content:
        content = content.replace(
            "from transformers import",
            "from transformers import GenerationMixin  # patched\nfrom transformers import",
            1
        )

# 2. Sınıf tanımını güncelle
content = content.replace(
    "class GPT2InferenceModel(GPT2PreTrainedModel):",
    "class GPT2InferenceModel(GPT2PreTrainedModel, GenerationMixin):"
)

if content != original:
    with open(file_path, "w", encoding="utf-8") as f:
        f.write(content)
    print("✅ gpt_inference.py patch başarıyla uygulandı!")
    print("   GPT2InferenceModel artık GenerationMixin'den kalıtım alıyor.")
else:
    print("⚠️  Değişiklik yapılmadı. Dosya zaten güncel olabilir.")
    print("   Manuel kontrol edin:", file_path)
